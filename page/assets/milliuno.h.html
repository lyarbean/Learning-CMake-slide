#ifndef _MILLIUNO_H_
#define _MILLIUNO_H_
#include <string>
#include <iostream>
#include <sstream>
#include <iomanip>
#include <map>
#include <cmath>
class Milliuno;
typedef std::map<std::string, Milliuno> RoleMap;
class Milliuno {
    public:
        Milliuno (double m = 0.0) {
            m_d = static_cast<int>(m * 1E3) * 1E-3;
        }

        Milliuno (int m) : m_d (m) {}

        Milliuno (const char* m) {
            if (!m) {
                m_d = 0.0;
                return;
            }
            std::stringstream ss;
            ss << m;
            ss >> m_d;
            m_d = static_cast<int>(m_d * 1E3) * 1E-3;
        }

        Milliuno (const std::string& m) {
            std::stringstream ss;
            ss << m;
            ss >> m_d;
            m_d = static_cast<int>(m_d * 1E3) * 1E-3;
        }

        Milliuno (const Milliuno& m) : m_d (m.m_d) {}

        operator double () const {
            return m_d;
        }

        operator std::string () const {
            std::ostringstream oss;
            oss << std::setprecision(3) << std::fixed << m_d;
            return oss.str();
        }

        Milliuno& operator=(const Milliuno& m) {
            if (this == &m) {
                return *this;
            }
            m_d = m.m_d;
            return *this;
        }

        Milliuno& operator=(double m) {
            m_d = static_cast<int>(m * 1E3) * 1E-3;
            return *this;
        }

        Milliuno& operator=(const std::string& m) {
            std::stringstream ss;
            ss << m;
            ss >> m_d;
            m_d = static_cast<int>(m_d * 1E3) * 1E-3;
            return *this;
        }

        Milliuno& operator=(const char* m) {
            if (!m) {
                m_d = 0.0;
                return *this;
            }
            std::stringstream ss;
            ss << m;
            ss >> m_d;
            m_d = static_cast<int>(m_d * 1E3) * 1E-3;
            return *this;
        }

        Milliuno& operator*=(double m) {
            m_d = static_cast<int>(m_d * m * 1E3) * 1E-3;
            return *this;
        }

        bool operator==(const Milliuno& rhs) {
            return fabs(m_d - rhs.m_d) < 1E-3;
        }

        bool operator==(const double& rhs) {
            return fabs(m_d - rhs) < 1E-3;
        }

        const std::string str() const {
            std::ostringstream oss;
            oss << std::setprecision(3) << std::fixed << m_d;
            return oss.str();
        }

        double val() const {
            return m_d;
        }

        friend std::ostream& operator<<(std::ostream& os, const Milliuno& m) {
            os << std::string(m);
            return os;
        }

        static std::map<std::string, double> FromRoleMap(const RoleMap& role_map) {
            std::map<std::string, double> rm;
            for (RoleMap::const_iterator it = role_map.begin(); it != role_map.end(); ++it) {
                rm[it->first] = it->second.val();
            }
            return rm;
        }
        static std::map<std::string, std::string> FromRoleMap2(const RoleMap& role_map) {
            std::map<std::string, std::string> rm;
            for (RoleMap::const_iterator it = role_map.begin(); it != role_map.end(); ++it) {
                rm[it->first] = it->second.str();
            }
            return rm;
        }
        static RoleMap ToRoleMap(const std::map<std::string, double>& role_map) {
            RoleMap rm;
            for (std::map<std::string, double>::const_iterator it = role_map.begin(); it != role_map.end(); ++it) {
                rm[it->first] = it->second;
            }
            return rm;
        }
    private:
        double m_d;
};

#endif // _MILLIUNO_H_
